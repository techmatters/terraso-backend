  $offset: Int,
  $before: String,
  $after: String,
  $first: Int,
  $last: Int,
  $name_Icontains: String,

  $slug: String,
  $slug_Icontains: String,
  $url_Icontains: String,
  $groups_Slug_Icontains: String,
  $groups_Id: UUID

  dataEntries(
    offset: $offset,
    before: $before,
    after: $after,
    first: $first,
    last: $last,

    description_Icontains: $description_Icontains,
    slug: $slug,
    slug_Icontains: $slug_Icontains,
    url_Icontains: $url_Icontains,

    groups_Id: $groups_Id

    pageInfo {
      ...PageInfoFragment
    }

      ...DataEntryNodeEdgeFragment

  }
